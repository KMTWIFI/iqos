ﺳﻴﺎﺳﺔ ﺍﻟﺨﺼﻮﺻﻴﺔ

ﺗﻬﺪﻑ ﺍﻟﺸﺮﻛﺔ ﻣﻦ ﺧﻼﻝ ﺳﻴﺎﺳﺔ ﺍﻟﺨﺼﻮﺻﻴﺔ ﺍﻟﺨﺎﺻﺔ ﺷﺮﺡ ﺍﻟﻤﻌﻠﻮﻣﺎﺕ ﺍﻟﺘﻲ ﻳﺘﻢ ﺟﻤﻌﻬﺎ ﻭ ﺍﻟﻬﺪﻑ ﻣﻦ ﺟﻤﻊ ﻫﺬﻩ ﺍﻟﻤﻌﻠﻮﻣﺎﺕ. ﻛﻤﺎ ﺗﻬﺪﻑ ﺍﻟﺸﺮﻛﺔ ﺍﻟﺘﺄﻛﺪ ﻣﻦ ﺃﻧﻚ ﻋﻠﻰ ﺩﺭﺍﻳﺔ ﺑﻄﺮﻕ ﺍﺳﺘﺨﺪﺍﻣﻨﺎ ﻟﻤﻌﻠﻮﻣﺎﺗﻚ ﻭﻣﺸﺎﺭﻛﺘﻬﺎ ﻭﺍﻟﺨﻴﺎﺭﺍﺕ ﺍﻟﻤﺘﺎﺣﺔ ﻟﻚ. ﻫﺪﻓﻨﺎ ﻫﻮ ﻛﺴﺐ ﺛﻘﺘﻚ ﻭﺍﻟﺤﻔﺎﻅ ﻋﻠﻴﻬﺎ ﺃﺛﻨﺎﺀ ﻗﻴﺎﻣﻚ ﺑﺰﻳﺎﺭﺓ ﻣﻮﺍﻗﻊ ﺍﻟﻮﻳﺐ ﻭﺍﻟﺘﻄﺒﻴﻘﺎﺕ ﺍﻟﺨﺎﺻﺔ ﺑﻤﻨﺘﺠﺎﺗﻨﺎ ﻭﻣﺎﺭﻛﺎﺗﻨﺎ.

ﺗﻮﺿﺢ ﻫﺬﻩ ﺍﻟﺴﻴﺎﺳﺔ ﻛﻴﻔﻴﺔ ﺗﻌﺎﻣﻠﻨﺎ ﻣﻊ ﺍﻟﻤﻌﻠﻮﻣﺎﺕ ﺍﻟﺸﺨﺼﻴﺔ ﻋﻠﻰ ﻣﻮﺍﻗﻊ ﺍﻟﻮﻳﺐ ("ﺍﻟﻤﻮﺍﻗﻊ")، ﺃﻭ ﺗﻄﺒﻴﻘﺎﺕ ﺍﻟﺠﻮﺍﻝ ﺃﻭ ﺍﻟﻮﺳﺎﺋﻂ ﺍﻻﺟﺘﻤﺎﻋﻴﺔ ("ﺍﻟﺘﻄﺒﻴﻘﺎﺕ")، ﻭﻛﺬﻟﻚ ﻓﻲ ﺗﻌﺎﻣﻠﻨﺎ ﻣﻊ ﺃﻱ ﺧﺪﻣﺎﺕ ﺃﺧﺮﻯ ﻧﻘﻮﻡ ﺑﺘﻀﻤﻴﻦ ﺭﺍﺑﻂ ﺃﻭ ﺍﻻﺷﺎﺭﺓ ﻟﻬﺬﻩ ﺍﻟﺴﻴﺎﺳﺔ (ﻳﺸﺎﺭ ﺍﻟﻴﻬﺎ، "ﺍﻟﺨﺪﻣﺎﺕ"). ﻻ ﺗﻌﺒﺮ ﻫﺬﻩ ﺍﻟﺴﻴﺎﺳﺔ ﻋﻦ ﻛﻴﻔﻴﺔ ﺍﻟﺘﻌﺎﻣﻞ ﻣﻊ ﺍﻟﻤﻮﺍﻗﻊ ﺃﻭ ﺍﻟﺘﻄﺒﻴﻘﺎﺕ ﺍﻟﺘﻲ ﻟﻢ ﺗﺘﻢ ﺍﻹﺷﺎﺭﺓ ﺇﻟﻴﻬﺎ ﻓﻰ ﻫﺬﺓ ﺍﻟﺴﻴﺎﺳﺔ

ﺍﺳﺘﺨﺪﺍﻣﻚ ﻟﻠﺘﻄﺒﻴﻘﺎﺕ ﻭ ﺍﻟﻤﻮﺍﻗﻊ ﺍﻟﺘﺎﺑﻌﺔ ﻟﻨﺎ ﻳﻌﺒﺮ ﻋﻦ ﻣﻮﺍﻓﻘﺘﻚ ﻋﻠﻰ ﻗﻴﺎﻣﻨﺎ ﺑﺠﻤﻊ ﺍﻟﻤﻌﻠﻮﻣﺎﺕ ﺍﻟﺨﺎﺻﺔ ﺑﻚ ﻭﺍﺳﺘﺨﺪﺍﻣﻬﺎ ﻭﻣﺸﺎﺭﻛﺘﻬﺎ ﻋﻠﻰ ﺍﻟﻨﺤﻮ ﺍﻟﻤﻨﺼﻮﺹ ﻋﻠﻴﻪ ﻓﻲ ﻫﺬﻩ ﺍﻟﺴﻴﺎﺳﺔ.

ﻣﺎ ﻫﻲ ﺍﻟﻤﻌﻠﻮﻣﺎﺕ ﺍﻟﺘﻲ ﻧﻘﻮﻡ ﺑﺠﻤﻌﻬﺎ؟

ﻧﻘﻮﻡ ﺑﺠﻤﻊ ﻣﻌﺮﻓﺎﺕ ﻭﻣﻌﻠﻮﻣﺎﺕ ﺍﻻﺗﺼﺎﻝ ﺍﻟﺨﺎﺻﺔ ﺑﻚ. ﻋﻠﻰ ﺳﺒﻴﻞ ﺍﻟﻤﺜﺎﻝ ﻭ ﻟﻴﺲ ﺍﻟﺤﺼﺮ، ﻗﺪ ﻧﺠﻤﻊ ﺍﺳﻤﻚ ﻭﻋﻨﻮﺍﻧﻚ ﺇﺫﺍ ﻗﻤﺖ ﺑﺎﻟﺘﺴﺠﻴﻞ ﻋﻠﻰ ﺃﺣﺪ ﻣﻮﻗﻌﻨﺎ ﺃﻭ ﺗﻄﺒﻴﻘﺎﺗﻨﺎ. ﻗﺪ ﻧﺠﻤﻊ ﺃﻳﻀًﺎ ﺭﻗﻢ ﻫﺎﺗﻔﻚ ﺃﻭ ﻋﻨﻮﺍﻥ ﺑﺮﻳﺪﻙ ﺍﻹﻟﻜﺘﺮﻭﻧﻲ ﺍﻟﺨﺎﺹ ﺑﻚ. ﻧﻘﻮﻡ ﺃﻳﻀًﺎ ﺑﺠﻤﻊ ﻣﻌﺮﻓﺎﺕ ﻋﺒﺮ ﺍﻹﻧﺘﺮﻧﺖ. ﻋﻠﻰ ﺳﺒﻴﻞ ﺍﻟﻤﺜﺎﻝ ﻭ ﻟﻴﺲ ﺍﻟﺤﺼﺮ، ﺇﺫﺍ ﺗﻔﺎﻋﻠﺖ ﻣﻌﻨﺎ ﻣﻦ ﺧﻼﻝ ﺃﺣﺪ ﻣﻮﺍﻗﻊ ﺍﻟﺘﻮﺍﺻﻞ ﺍﻻﺟﺘﻤﺎﻋﻲ، ﻓﻘﺪ ﻧﺠﻤﻊ ﺍﺳﻢ ﻣﺴﺘﺨﺪﻡ ﺍﻟﻮﺳﺎﺋﻂ ﺍﻻﺟﺘﻤﺎﻋﻴﺔ ﺍﻟﺨﺎﺹ ﺑﻚ.

ﻧﻘﻮﻡ ﺑﺠﻤﻊ ﺍﻟﻤﻌﻠﻮﻣﺎﺕ ﺍﻟﺘﻲ ﺗﻘﻮﻡ ﺑﺈﺩﺧﺎﻟﻬﺎ ﺃﻭ ﻣﻮﺍﻓﺎﺗﻨﺎ ﺑﻬﺎ ﺃﻭ ﺍﻟﺘﻰ ﺗﻔﻮﺿﻨﺎ ﺑﺠﻤﻌﻬﺎ

ﺟﻤﻊ ﺍﻟﺒﻴﺎﻧﺎﺕ ﺍﻟﺸﺨﺼﻴﺔ:

  • ﻳﻮﺍﻓﻖ ﺍﻟﻌﻤﻴﻞ ﺍﻥ ﺑﻴﺎﻧﺎﺗﻪ ﺍﻟﻤﻘﺪﻣﺔ ﻟﻠﺸﺮﻛﺔ ﻳﻤﻜﻦ ﺗﺴﺠﻴﻠﻬﺎ ﺍﻭ ﺍﺳﺘﺨﺪﺍﻣﻬﺎ ﻭﻳﻤﻜﻦ ﻣﻨﺤﻬﺎ ﻟﻌﻤﻼﺀ ﺍﻟﺸﺮﻛﺔ

ﻛﻴﻒ ﻧﺠﻤﻊ ﻣﻌﻠﻮﻣﺎﺕ؟

ﻧﻘﻮﻡ ﺑﺠﻤﻊ ﺍﻟﻤﻌﻠﻮﻣﺎﺕ ﺑﻄﺮﻳﻘﺔ ﻣﺒﺎﺷﺮﺓ ﻣﻦ ﺧﻼﻟﻚ . ﻧﻘﻮﻡ ﺑﺠﻤﻊ ﻣﻌﻠﻮﻣﺎﺕ ﻣﻦ ﺧﻼﻟﻚ ﻋﻨﺪﻣﺎ:

  • ﺗﻘﻮﻡ ﺑﺎﻟﺸﺮﺍﺀ ﻣﻌﻨﺎ
  • ﺗﻘﻮﻡ ﺑﺎﻟﺘﺴﺠﻴﻞ ﻓﻲ ﺃﺣﺪ ﻣﻮﺍﻗﻌﻨﺎ ﺃﻭ ﺗﻄﺒﻴﻘﺎﺗﻨﺎ
  • ﺍﻻﻧﻀﻤﺎﻡ ﺇﻟﻰ ﺑﺮﺍﻣﺞ ﺍﻟﻮﻻﺀ ﺍﻟﻤﺘﺎﺣﺔ ﻟﺪﻳﻨﺎ
  • ﺸﺘﺮﻙ ﻓﻲ ﺍﻟﻨﺸﺮﺍﺕ ﺍﻹﺧﺒﺎﺭﻳﺔ ﻭﺍﻹﺧﻄﺎﺭﺍﺕ ﺍﻷﺧﺮﻯ ﺍﻟﺨﺎﺻﺔ ﺑﻨﺎ ﻋﺒﺮ ﺍﻟﺒﺮﻳﺪ ﺍﻹﻟﻜﺘﺮﻭﻧﻲ
  • ﺗﺸﺎﺭﻙ ﻓﻲ ﺍﻟﻌﺮﻭﺽ ﺍﻟﺘﺮﻭﻳﺠﻴﺔ ﻣﺜﻞ ﺍﻟﻤﺴﺎﺑﻘﺎﺕ ﻭﻓﻘﺎ ﻟﻠﺸﺮﻭﻁ ﻭ ﺍﻷﺣﻜﺎﻡ ﺍﻟﺨﺎﺻﺔ ﺑﻜﻞ ﻋﺮﺽ ﺃﻭ ﻣﺴﺎﺑﻘﺔ
  • ﺗﻤﻠﺊ ﺍﻻﺳﺘﺒﻴﺎﻥ
  • ﺗﺘﺼﻞ ﺑﻨﺎ ، ﻋﻠﻰ ﺳﺒﻴﻞ ﺍﻟﻤﺜﺎﻝ ﻣﻦ ﺧﻼﻝ ﻣﻮﺍﻗﻌﻨﺎ ﻭﺗﻄﺒﻴﻘﺎﺗﻨﺎ ﺃﻭ ﺍﻟﺘﻮﺍﺻﻞ ﻣﻊ ﻓﺮﻳﻖ ﻋﻼﻗﺎﺕ ﺍﻟﻤﺴﺘﻬﻠﻚ ﻟﺪﻳﻨﺎ ﻭ

ﻣﻦ ﺧﻼﻝ ﺃﻯ ﻃﺮﻳﻘﺔ ﺃﺧﺮﻯ ﺗﻘﻮﻡ ﻣﻦ ﺧﻼﻟﻬﺎ ﺑﺘﻘﺪﻳﻢ ﺍﻟﻤﻌﻠﻮﻣﺎﺕ ﻟﻨﺎ ﺃﻭ ﺗﻔﻮﻳﻀﻨﺎ ﺑﺠﻤﻌﻬﺎ

ﻛﻴﻒ ﻧﺴﺘﺨﺪﻡ ﺍﻟﻤﻌﻠﻮﻣﺎﺕ ﺍﻟﺘﻲ ﻧﺠﻤﻌﻬﺎ؟

ﻧﺴﺘﺨﺪﻡ ﺍﻟﻤﻌﻠﻮﻣﺎﺕ ﻟﻤﺴﺎﻋﺪﺗﻨﺎ ﻋﻠﻰ ﺗﻮﻓﻴﺮ ﺍﻟﻤﻨﺘﺠﺎﺕ ﻭﺍﻟﺨﺪﻣﺎﺕ ﻟﻚ ﺑﺸﻜﻞ ﺃﻓﻀﻞ.

ﻧﺴﺘﺨﺪﻡ ﺍﻟﻤﻌﻠﻮﻣﺎﺕ ﻟﺘﺤﺴﻴﻦ ﻣﻨﺘﺠﺎﺗﻨﺎ ﻭﺧﺪﻣﺎﺗﻨﺎ.

ﻧﺤﻦ ﻧﺴﺘﺨﺪﻡ ﺍﻟﻤﻌﻠﻮﻣﺎﺕ ﻷﻏﺮﺍﺽ ﺍﻟﺘﺴﻮﻳﻖ

ﻧﺴﺘﺨﺪﻡ ﺍﻟﻤﻌﻠﻮﻣﺎﺕ ﻟﻠﺤﺼﻮﻝ ﻋﻠﻰ ﺗﻌﻠﻴﻘﺎﺕ ﻣﻨﻚ ﺣﻮﻝ ﺃﺩﺍﺋﻨﺎ ﻭﻷﻏﺮﺍﺽ ﺗﺘﻌﻠﻖ ﺑﺎﻟﻤﻌﺎﻣﻼﺕ ﺍﻷﺧﺮﻯ ﻣﺜﻞ ﻋﻤﻠﻴﺔ ﺍﻟﺘﺴﺠﻴﻞ ﻋﻠﻲ ﻣﻮﻗﻌﻨﺎ ﻭﺍﻟﻤﻌﺎﻣﻼﺕ ﺍﻟﻤﻤﺎﺛﻠﺔ ﻟﺬﻟﻚﻧﺤﻦ ﻧﺴﺘﺨﺪﻡ ﺍﻟﻤﻌﻠﻮﻣﺎﺕ ﻛﻤﺎ ﻳﺴﻤﺢ ﻟﻨﺎ ﺍﻟﻘﺎﻧﻮﻥ ﻭﻛﻤﺎ ﻫﻮ ﻣﺬﻛﻮﺭ ﻓﻲ ﻫﺬﻩ ﺍﻟﺴﻴﺎﺳﺔ.

ﻧﺤﻦ ﻧﺸﺎﺭﻙ ﺍﻟﻤﻌﻠﻮﻣﺎﺕ ﻣﻊ ﺷﺮﻛﺎﺋﻨﺎ ﺍﻟﺘﺠﺎﺭﻳﻴﻦ ﻭﺍﻟﻤﻮﺭﺩﻳﻦ ﺍﻟﺨﺎﺭﺟﻴﻴﻦ. ﻋﻠﻰ ﺳﺒﻴﻞ ﺍﻟﻤﺜﺎﻝ ، ﻗﺪ ﻧﺸﺎﺭﻙ ﺍﻟﻤﻌﻠﻮﻣﺎﺕ ﻣﻊ ﺷﺮﻛﺌﻨﺎ ﻓﻰ ﺣﺎﻻﺕ ﺍﻟﺘﺮﻭﻳﺞ ﺍﻟﻤﺸﺘﺮﻙ. ﺇﺫﺍ ﻗﻤﺖ ﺑﺎﻟﺘﺴﺠﻴﻞ ﻓﻲ ﻋﺮﺽ ﺗﺮﻭﻳﺠﻲ ﻣﺸﺘﺮﻙ ، ﻓﻘﺪ ﻳﺘﻢ ﺟﻤﻊ ﻣﻌﻠﻮﻣﺎﺗﻚ ﻣﻦ ﻗﺒﻠﻨﺎ ﻭﻣﻦ ﻗﺒﻞ ﺍﻟﻐﻴﺮ ﻭ ﻗﺪ ﻳﺸﻤﻞ ﻫﺬﺍ ﺗﺠﺎﺭ ﺍﻟﺘﺠﺰﺋﺔ. ﺳﻴﺘﻢ ﺍﺳﺘﺨﺪﺍﻡ ﻣﻌﻠﻮﻣﺎﺗﻚ ﻣﻦ ﻗﺒﻠﻨﺎ ﻛﻤﺎ ﻫﻮ ﻣﻮﺿﺢ ﻓﻲ ﻫﺬﻩ ﺍﻟﺴﻴﺎﺳﺔ ﺃﻭ ﺑﻮﺍﺳﻄﺔ ﺷﺮﻛﺎﺀ ﺍﻟﺘﺮﻭﻳﺞ ﻛﻤﺎ ﻫﻮ ﻣﻮﺿﺢ ﻓﻲ ﺳﻴﺎﺳﺎﺗﻬﻢ. ﻳﺠﻮﺯ ﻟﻨﺎ ﺃﻳﻀًﺎ ﻣﺸﺎﺭﻛﺔ ﺍﻟﻤﻌﻠﻮﻣﺎﺕ ﻣﻊ ﺍﻟﺸﺮﻛﺎﺕ ﺍﻟﺘﻲ ﺗﻌﻤﻞ ﻛﺒﺎﺋﻌﻴﻦ ﻟﻨﺎ ﺃﻭ ﺃﻃﺮﺍﻑ ﺍﺧﺮﻱ ﺗﺆﺩﻱ ﺧﺪﻣﺎﺕ ﻧﻴﺎﺑﺔ ﻋﻨﺎ. ﻗﺪ ﻳﺴﺎﻋﺪﻭﻧﻨﺎ ﻓﻲ ﺗﻘﺪﻳﻢ ﺍﻟﻤﻨﺘﺠﺎﺕ ﺃﻭ ﺍﻟﺨﺪﻣﺎﺕ ﻟﻚ. ﻳﺠﻮﺯ ﻟﻨﺎ ﻣﺸﺎﺭﻛﺔ ﺍﻟﻤﻌﻠﻮﻣﺎﺕ ﻣﻊ ﺍﻟﺸﺮﻛﺎﺕ ﺍﻟﺘﻲ ﺗﺪﻳﺮ ﻣﻮﺍﻗﻌﻨﺎ ﻭﺗﻄﺒﻴﻘﺎﺗﻨﺎ ﺃﻭ ﺍﻟﺘﻲ ﺗﺪﻳﺮ ﻋﺮﻭﺿﻨﺎ ﺍﻟﺘﺮﻭﻳﺠﻴﺔ.

ﺳﻮﻑ ﻧﺸﺎﺭﻙ ﺍﻟﻤﻌﻠﻮﻣﺎﺕ ﺇﺫﺍ ﺍﻋﺘﻘﺪﻧﺎ ﺃﻧﻪ ﻳﺠﺐ ﻋﻠﻴﻨﺎ ﺍﻻﻣﺘﺜﺎﻝ ﻟﻠﻘﺎﻧﻮﻥ ﺃﻭ ﻟﺤﻤﺎﻳﺔ ﺷﺮﻛﺘﻨﺎ. ﻋﻠﻰ ﺳﺒﻴﻞ ﺍﻟﻤﺜﺎﻝ ، ﺳﻮﻑ ﻧﺸﺎﺭﻙ ﺍﻟﻤﻌﻠﻮﻣﺎﺕ ﻟﻠﺮﺩ ﻋﻠﻰ ﺃﻣﺮ ﻣﺤﻜﻤﺔ ﺃﻭ ﺃﻣﺮ ﺍﺳﺘﺪﻋﺎﺀ. ﻳﺠﻮﺯ ﻟﻨﺎ ﻣﺸﺎﺭﻛﺔ ﺍﻟﻤﻌﻠﻮﻣﺎﺕ ﺇﺫﺍ ﻃﻠﺒﺖ ﺫﻟﻚ ﻭﻛﺎﻟﺔ ﺣﻜﻮﻣﻴﺔ ﺃﻭ ﻫﻴﺌﺔ ﺗﺤﻘﻴﻖ. ﻭﻫﺬﺍ ﻳﺸﻤﻞ ﺳﻠﻄﺎﺕ ﺇﻧﻔﺎﺫ ﺍﻟﻘﺎﻧﻮﻥ ﺃﻭ ﺍﻟﺴﻠﻄﺎﺕ ﺍﻟﺘﻨﻈﻴﻤﻴﺔ ﻓﻲ ﺍﻟﻮﻻﻳﺎﺕ ﺍﻟﻤﺘﺤﺪﺓ ﻭﻏﻴﺮﻫﺎ ﻣﻦ ﺍﻟﺒﻼﺩ. ﻗﺪ ﻧﺸﺎﺭﻙ ﺃﻳﻀًﺎ ﺍﻟﻤﻌﻠﻮﻣﺎﺕ ﻟﻠﻤﺴﺎﻋﺪﺓ ﻓﻲ ﺣﻤﺎﻳﺘﻚ ﻭﺣﻤﺎﻳﺔ ﻣﻮﺍﻗﻌﻨﺎ ﻭﺗﻄﺒﻴﻘﺎﺗﻨﺎ ﻭﺷﺮﻛﺘﻨﺎ ﻭﻋﻤﻼﺋﻨﺎ ﻭﻣﺴﺘﻬﻠﻜﻴﻨﺎ. ﻳﺠﻮﺯ ﻟﻨﺎ ﺃﻳﻀًﺎ ﻣﺸﺎﺭﻛﺔ ﺍﻟﻤﻌﻠﻮﻣﺎﺕ ﺇﺫﺍ ﻛﻨﺎ ﻧﺤﻘﻖ ﻓﻲ ﺍﺣﺘﻴﺎﻝ ﻣﺤﺘﻤﻞ.

ﻳﺠﻮﺯ ﻟﻠﻐﻴﺮ ﺟﻤﻊ ﺍﻟﻤﻌﻠﻮﻣﺎﺕ ﻋﻨﺪ ﺍﺳﺘﺨﺪﺍﻣﻚ ﻟﻠﺨﺪﻣﺎﺕ. ﻳﺠﻮﺯ ﻷﻱ ﻃﺮﻑ ﻣﻦ ﺍﻟﻐﻴﺮﺍﻟﺬﻱ ﻳﻘﺪﻡ ﺇﻋﻼﻧﺎﺕ ﻭ / ﺃﻭ ﻣﺤﺘﻮﻯ ﻭ / ﺃﻭ ﻭﻇﺎﺋﻒ ﻟﺨﺪﻣﺎﺗﻨﺎ، ﺟﻤﻊ ﻣﻌﻠﻮﻣﺎﺕ ﻣﻦ ﻫﺬﻩ ﺍﻟﺨﺪﻣﺎﺕ ﻣﻦ ﺧﻼﻝ ﺃﺩﻭﺍﺕ ﺍﻟﺘﺘﺒﻊ ﻣﺜﻞ ﻣﻠﻔﺎﺕ ﺗﻌﺮﻳﻒ ﺍﻻﺭﺗﺒﺎﻁ ﻟﻠﻤﺘﺼﻔﺢ ﻭﻣﻌﺮﻓﺎﺕ ﺍﻟﺠﻬﺎﺯ ﺍﻟﻤﺤﻤﻮﻝ. ﻋﻠﻰ ﺳﺒﻴﻞ ﺍﻟﻤﺜﺎﻝ ، ﻗﺪ ﻳﻜﻮﻥ ﻟﻠﻐﻴﺮﺗﻮﺍﺟﺪ ﻓﻲ ﺑﻌﺾ ﺧﺪﻣﺎﺗﻨﺎ ، ﻭﻗﺪ ﻳﺘﻢ ﺟﻤﻊ ﻫﺬﻩ ﺍﻟﻤﻌﻠﻮﻣﺎﺕ ﺑﻤﺮﻭﺭ ﺍﻟﻮﻗﺖ ﻭﺩﻣﺠﻬﺎ ﻣﻊ ﺍﻟﻤﻌﻠﻮﻣﺎﺕ ﺍﻟﺘﻲ ﻳﺘﻢ ﺟﻤﻌﻬﺎ ﻋﺒﺮ ﻣﻮﺍﻗﻊ ﺍﻟﻮﻳﺐ ﻭﺍﻟﺨﺪﻣﺎﺕ ﺍﻟﻤﺨﺘﻠﻔﺔ ﻋﺒﺮ ﺍﻹﻧﺘﺮﻧﺖ.

ﻳﺠﻮﺯ ﻟﻨﺎ ﻣﺸﺎﺭﻛﺔ ﺍﻟﻤﻌﻠﻮﻣﺎﺕ ﻣﻊ ﺍﻟﻜﻴﺎﻥ ﺍﻟﺬﻱ ﻳﺨﻠﻔﻨﺎ ﻓﻲ ﻛﻞ ﺃﻋﻤﺎﻟﻨﺎ ﺃﻭ ﺟﺰﺀ ﻣﻨﻬﺎ. ﻋﻠﻰ ﺳﺒﻴﻞ ﺍﻟﻤﺜﺎﻝ ، ﺇﺫﺍ ﺗﻢ ﺑﻴﻊ ﺟﺰﺀ ﻣﻦ ﺃﻋﻤﺎﻟﻨﺎ ﺃﻭ ﻣﺸﺎﺭﻛﺘﻪ ﻓﻲ ﻋﻤﻠﻴﺔ ﺍﻧﺪﻣﺎﺝ ﺃﻭ ﺗﻮﺣﻴﺪ ﺃﻭ ﺑﻴﻊ ﺃﺻﻮﻝ ﺃﻭ ﻓﻲ ﺣﺎﻟﺔ ﺇﻓﻼﺱ ﻏﻴﺮ ﻣﺤﺘﻤﻠﺔ ، ﻓﻘﺪ ﻧﻘﻮﻡ ﺑﺘﻀﻤﻴﻦ ﻣﻌﻠﻮﻣﺎﺗﻚ ﻛﺠﺰﺀ ﻣﻦ ﺗﻠﻚ ﺍﻟﺼﻔﻘﺔ.

ﻟﺪﻳﻚ ﺧﻴﺎﺭﺍﺕ ﻣﻌﻴﻨﺔ ﺣﻮﻝ ﻛﻴﻔﻴﺔ ﺍﺳﺘﺨﺪﺍﻣﻨﺎ ﻟﻤﻌﻠﻮﻣﺎﺗﻚ

ﻳﻤﻜﻨﻚ ﺇﻟﻐﺎﺀ ﺍﻻﺷﺘﺮﺍﻙ ﻓﻲ ﺍﻟﺘﺴﻮﻳﻖ ﻋﺒﺮ ﺍﻟﺒﺮﻳﺪ ﺍﻹﻟﻜﺘﺮﻭﻧﻲ. ﻹﻳﻘﺎﻑ ﺗﻠﻘﻲ ﺭﺳﺎﺋﻞ ﺍﻟﺒﺮﻳﺪ ﺍﻹﻟﻜﺘﺮﻭﻧﻲ ﺍﻟﺘﺮﻭﻳﺠﻴﺔ ﻣﻨﺎ ، ﺍﺗﺒﻊ ﺍﻟﺘﻌﻠﻴﻤﺎﺕ ﺍﻟﻮﺍﺭﺩﺓ ﻓﻲ ﺃﻱ ﺭﺳﺎﻟﺔ ﺑﺮﻳﺪ ﺇﻟﻜﺘﺮﻭﻧﻲ ﺗﺮﻭﻳﺠﻴﺔ ﺗﺘﻠﻘﺎﻫﺎ ﻣﻨﺎ. ﺇﺫﺍ ﺍﺧﺘﺮﺕ ﻋﺪﻡ ﺗﻠﻘﻲ ﺭﺳﺎﺋﻞ ﺍﻟﺒﺮﻳﺪ ﺍﻹﻟﻜﺘﺮﻭﻧﻲ ﺍﻟﺘﺴﻮﻳﻘﻲ ، ﻓﻘﺪ ﺗﺴﺘﻤﺮ ﻓﻲ ﺗﻠﻘﻲ ﺭﺳﺎﺋﻞ ﺑﺮﻳﺪ ﺇﻟﻜﺘﺮﻭﻧﻲ ﻏﻴﺮ ﺗﺴﻮﻳﻘﻴﺔ ﻣﻨﺎ ﺣﻴﺜﻤﺎ ﻳﺴﻤﺢ ﺍﻟﻘﺎﻧﻮﻥ ﺑﺬﻟﻚ ﻛﻤﺎ ﻳﻤﻜﻨﻚ ﻛﺬﻟﻚ ﺍﺧﺘﻴﺎﺭ ﻋﺪﻡ ﺗﻠﻘﻲ ﺭﺳﺎﺋﻞ ﺑﺮﻳﺪ ﺇﻟﻜﺘﺮﻭﻧﻲ ﻏﻴﺮ ﺗﺴﻮﻳﻘﻴﺔ ﻣﻨﺎ .

ﺗﺪﺍﺑﻴﺮ ﺃﻣﻨﻴﺔ
